    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const ctx = document.getElementById('skillsChart');
            const mobileMenuButton = document.getElementById('mobile-menu-button');
            const mobileMenu = document.getElementById('mobile-menu');
            const generateButton = document.getElementById('generate-summary-button');
            const projectDescription = document.getElementById('project-description');
            const summaryOutput = document.getElementById('summary-output');
            const summaryText = document.getElementById('summary-text');
            const loadingIndicator = document.getElementById('loading-indicator');

            if (ctx) {
                new Chart(ctx, {
                    type: 'radar',
                    data: {
                        labels: [
                            'Programming (C#, JS, SQL)', 
                            'Frameworks (Odoo, Salesforce)', 
                            'BI & Cloud (Power BI, Azure)', 
                            'Design (Figma, Visio, draw.io)', 
                            'Mgt Tools (Jira, MS Project)'
                        ],
                        datasets: [{
                            label: 'Proficiency',
                            data: [8, 9, 8, 8, 9],
                            fill: true,
                            backgroundColor: 'rgba(74, 144, 226, 0.2)',
                            borderColor: 'rgb(74, 144, 226)',
                            pointBackgroundColor: 'rgb(74, 144, 226)',
                            pointBorderColor: '#fff',
                            pointHoverBackgroundColor: '#fff',
                            pointHoverBorderColor: 'rgb(74, 144, 226)'
                        }]
                    },
                    options: {
                        maintainAspectRatio: false,
                        elements: {
                            line: {
                                borderWidth: 2
                            }
                        },
                        scales: {
                            r: {
                                angleLines: {
                                    display: true
                                },
                                suggestedMin: 0,
                                suggestedMax: 10,
                                grid: {
                                    color: 'rgba(0, 0, 0, 0.05)'
                                },
                                pointLabels: {
                                    font: {
                                        size: 12,
                                        family: "'Inter', sans-serif"
                                    },
                                    color: '#333D44'
                                },
                                ticks: {
                                    display: false,
                                    stepSize: 2
                                }
                            }
                        },
                        plugins: {
                            legend: {
                                display: false
                            },
                             tooltip: {
                                callbacks: {
                                    label: function(context) {
                                        let label = context.dataset.label || '';
                                        if (label) {
                                            label += ': ';
                                        }
                                        if (context.parsed.r !== null) {
                                            label += context.parsed.r + ' / 10';
                                        }
                                        return label;
                                    }
                                }
                            }
                        }
                    }
                });
            }
            
            mobileMenuButton.addEventListener('click', () => {
                mobileMenu.classList.toggle('hidden');
            });
            
            document.querySelectorAll('#mobile-menu a, .hidden.md\\:flex a').forEach(anchor => {
                anchor.addEventListener('click', function (e) {
                    if (this.getAttribute('href').startsWith('#')) {
                        e.preventDefault();
                        if(!mobileMenu.classList.contains('hidden')){
                           mobileMenu.classList.add('hidden');
                        }
                        document.querySelector(this.getAttribute('href')).scrollIntoView({
                            behavior: 'smooth'
                        });
                    }
                });
            });

            generateButton.addEventListener('click', async () => {
                const description = projectDescription.value.trim();
                if (!description) return;

                loadingIndicator.classList.remove('hidden');
                summaryOutput.classList.add('hidden');

                const systemPrompt = "You are a world-class professional resume builder. Your task is to take a raw description of a project or task and transform it into a concise, impactful, and professionally-worded resume bullet point or short paragraph. Focus on quantifiable results and key skills. The output should be a single, well-formatted paragraph.";
                const userQuery = `Convert the following project description into a professional resume summary: "${description}"`;

                const apiKey = "";
                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;

                const payload = {
                    contents: [{ parts: [{ text: userQuery }] }],
                    systemInstruction: { parts: [{ text: systemPrompt }] },
                };

                let retries = 0;
                const maxRetries = 3;
                let response;

                while (retries < maxRetries) {
                    try {
                        response = await fetch(apiUrl, {
                            method: 'POST',
                            headers: { 'Content-Type': 'application/json' },
                            body: JSON.stringify(payload)
                        });

                        if (response.ok) {
                            break;
                        } else if (response.status === 429) {
                            retries++;
                            const delay = Math.pow(2, retries) * 1000;
                            await new Promise(res => setTimeout(res, delay));
                        } else {
                            throw new Error(`API returned status code: ${response.status}`);
                        }
                    } catch (error) {
                        console.error('Fetch error:', error);
                        retries++;
                        if (retries >= maxRetries) {
                            summaryText.textContent = 'Failed to generate summary. Please try again later.';
                            summaryOutput.classList.remove('hidden');
                            loadingIndicator.classList.add('hidden');
                            return;
                        }
                    }
                }

                if (!response || !response.ok) {
                    summaryText.textContent = 'Failed to generate summary after multiple retries. Please check your network connection and try again.';
                    summaryOutput.classList.remove('hidden');
                    loadingIndicator.classList.add('hidden');
                    return;
                }

                const result = await response.json();
                const generatedText = result?.candidates?.[0]?.content?.parts?.[0]?.text;

                if (generatedText) {
                    summaryText.textContent = generatedText;
                } else {
                    summaryText.textContent = 'Could not generate a summary. The model may have returned an empty response.';
                }

                loadingIndicator.classList.add('hidden');
                summaryOutput.classList.remove('hidden');
            });
        });
    </script>
